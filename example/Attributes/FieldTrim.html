How to use the [FieldTrim] attribute (useful for fixed length records)

RecordClass.cs
{% highlight csharp %} [FixedLengthRecord()] public class Customer { [FieldFixedLength(5)] public int CustId; [FieldFixedLength(30)] [FieldTrim(TrimMode.Both)] public string Name; [FieldFixedLength(8)] [FieldConverter(ConverterKind.Date, "ddMMyyyy")] public DateTime AddedDate; } {% endhighlight %}
