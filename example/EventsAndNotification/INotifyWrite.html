Report layout.cs
{% highlight csharp %} [FixedLengthRecord] [IgnoreEmptyLines] public class OrdersFixed :INotifyWrite { [FieldFixedLength(7)] public int OrderID; [FieldFixedLength(8)] public string CustomerID; [FieldFixedLength(8)] public DateTime OrderDate; [FieldFixedLength(11)] public decimal Freight; public void BeforeWrite(BeforeWriteEventArgs e) { // We only want clients with large frieght values if (Freight < 40) e.SkipThisRecord = true; } public void AfterWrite(AfterWriteEventArgs e) { // Hide a line if (CustomerID.Trim() == "HANAR") e.RecordLine = "-- Insufficient Access"; } } {% endhighlight %}
